    // Edit button click event
    $('.editPackagesBtn').on('click', function () {
      // Get the data attributes from the clicked button
      var packageName = $(this).data('package-name');
      var packagePrice = $(this).data('package-price');

      // Check if data is being retrieved correctly
      console.log('Package Name:', packageName);
      console.log('Package Price:', packagePrice);

      // Set the values in the form fields in the modal
      $('#editPackageName').val(packageName);
      $('#editPrice').val(packagePrice);

      // Show the modal
      $('#editPackagesModal').modal('show');
    });

    // Prevent form submission if any
    // $('#editPackageForm').on('submit', function(event) {
    //     event.preventDefault(); // Prevent form from submitting
    // });

    $('#updatePackageBtn').on('click', function () {
      // var packageId = $('#editPackageId').val();
      var packageName = $('#editPackageName').val();
      var packagePrice = $('#editPrice').val();
      var userId = <?= json_encode($userId) ?>; // Ensure the userId is properly passed


      $.ajax({
        url: '<?= route_to('resellers.updatePackage') ?>', // The named route for your update method
        method: 'POST',
        headers: {
          'X-CSRF-TOKEN': csrfToken // Include CSRF token in the headers
        },
        data: {

          userId: userId,
          packageName: packageName,
          price: packagePrice,
          // Include CSRF token in the request
        },
        success: function (response) {
          if (response.status === 'success') {
            location.reload(); // Reload the page or update the table dynamically
          } else {
            alert('Failed to update package');
          }
        },
        error: function () {
          alert('An error occurred');
        }
      });
    });



  });
